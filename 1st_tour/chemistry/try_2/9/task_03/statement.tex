            \item Сульфат меди и вода
            \item Вода и этиловый спирт
            \item Поваренная соль и сера
        \end{enumerate}

            \item[а.] Растворение + фильтрование + выпаривание
            \item[б.] Выпаривание + конденсация паров
            \item[в.] Дистилляция
        \end{enumerate}
